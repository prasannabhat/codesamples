    def read
      begin
        str = @source.readline(@line_break)
        str = decode(str) if @to_utf and str 
        @buffer << str
        if not @to_utf and @buffer.respond_to? :force_encoding
          @buffer.force_encoding Encoding::UTF_8
        end
      rescue Exception, NameError
        @source = nil
      end
    end
